 public class Pagamento {

     private final LocalDate date;
     private final PaymentTypeEnum type;

         this.value = value;
         this.date = date;
         this.type = type;
     }

     }

     }

     }
 
     @Override
     public int hashCode() {
         return Objects.hash(value, date, type);
     }

     public String toString() {
         return "Pagamento{" +

                 ", type=" + type +
                 '}';
